INICIO

CARACTERE exec, response;
TEXTO studentName, subject;
INTEIRO docId, studentId, subjectCount;
REAL subjectAverage ← 0., totalAverage ← 0., averageSum;

FACA

    ESCREVA "SEJA BEM-VINDO AO SISTEMA DE MÉDIAS E MATRÍCULAS";
    ESCREVA "Deseja iniciar um novo aluno? Digite C para continuar ou S para sair";
    LEIA exec;

    SE (exec == 'S' OU exec == 's') ENTÃO
        PARE;
    FIMSE

    ESCREVA "Para iniciar, digite o Nome do Aluno:";
    LEIA studentName;

    ESCREVA "Agora digite o numero do Documento de Identificacao do aluno (somente numeros):";
    LEIA docId;

    ESCREVA "Por fim, digite o Numero de Matricula do aluno (somente numeros):";
    LEIA studentId;

    // Reiniciar contadores para novo aluno
    subjectCount ← 0;
    averageSum ← 0.;

    FACA
        ESCREVA "Insira o nome da Disciplina cursada:";
        LEIA subject;

        ESCREVA "Insira a Media do aluno nesta Disciplina:";
        LEIA subjectAverage;

        SE subjectAverage ≥ 0 ENTÃO
            subjectCount ← subjectCount + 1;
            averageSum ← averageSum + subjectAverage;
        FIMSE

    ENQUANTO subjectAverage >= 0;

    totalAverage ← averageSum / subjectCount;

    ESCREVA "MEDIA GERAL DO ALUNO: ", totalAverage;

    SE totalAverage ≥ 7 ENTAO
        ESCREVA "O aluno foi APROVADO. Escolha uma das 5 disciplinas avancadas para cursar:";
        ESCREVA "1-DISCIPLINA AVANÇADA I";
        ESCREVA "2-DISCIPLINA AVANÇADA II";
        ESCREVA "3-DISCIPLINA AVANÇADA III";
        ESCREVA "4-DISCIPLINA AVANÇADA IV";
        ESCREVA "5-DISCIPLINA AVANÇADA V";
    SENAO
        ESCREVA "O aluno foi REPROVADO. O aluno deseja repetir a disciplina? (s/n)";
        LEIA response

        SE response == 's' ENTAO
            ESCREVA "Escolha uma das 4 disciplinas para cursar:\n1-DISCIPLINA BASICA I\n2-DISCIPLINA BASICA II\n3-DISCIPLINA BASICA III\n4-DISCIPLINA BASICA IV";
            ESCREVA "1-DISCIPLINA BASICA I";
            ESCREVA "2-DISCIPLINA BASICA II";
            ESCREVA "3-DISCIPLINA BASICA III";
            ESCREVA "4-DISCIPLINA BASICA IV";
        SENAO
            ESCREVA "Escolha uma das 5 disciplinas para cursar:\n1-DISCIPLINA BASICA I\n2-DISCIPLINA BASICA II\n3-DISCIPLINA BASICA III\n4-DISCIPLINA BASICA IV";
            ESCREVA "1-DISCIPLINA ALTERNATIVA I";
            ESCREVA "2-DISCIPLINA ALTERNATIVA II";
            ESCREVA "3-DISCIPLINA ALTERNATIVA III";
            ESCREVA "4-DISCIPLINA ALTERNATIVA IV";
            ESCREVA "5-DISCIPLINA ALTERNATIVA V";
        FIMSENAO
    FIMSESNAO

ENQUANTO (exec = 'S' OU exec = 's');

FIM
